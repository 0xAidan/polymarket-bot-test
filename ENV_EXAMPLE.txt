# ============================================================
# Signature Type - IMPORTANT! Must match how you created your Polymarket account
# ============================================================
# 0 = EOA wallet (standalone EOA)
# 1 = POLY_PROXY — use for standard Polymarket proxy wallets (most users)
# 2 = POLY_GNOSIS_SAFE (existing Gnosis Safe flow)
# 3 = POLY_1271 (deposit wallet flow for new API users)
#
# HOW TO KNOW WHICH TO USE:
# 1. Go to polymarket.com/settings
# 2. Look for "Proxy Wallet" address
# 3. If you see a proxy wallet address different from your signer address → Use type 1
# 4. If you use a Safe wallet flow → Use type 2
# 5. If you are on deposit wallet flow (new API users) → Use type 3
# 6. If no proxy/safe/deposit wallet exists → Use type 0
#
# If you're getting "invalid signature" errors, you likely have the wrong type!
POLYMARKET_SIGNATURE_TYPE=0

# Funder/Proxy Wallet Address - IMPORTANT FOR BALANCE DISPLAY!
# This is where Polymarket holds your USDC. To find it:
# 1. Go to polymarket.com/settings
# 2. Look for "Proxy Wallet" or "Trading Wallet" address
# 3. Copy that address here
# 
# If you see $0 balance in the dashboard but have funds on Polymarket,
# you likely need to set this to your proxy wallet address.
# 
# You can also find it by going to polygonscan.com with your EOA address
# and looking for token transfers to your Polymarket account.
POLYMARKET_FUNDER_ADDRESS=0x_your_proxy_wallet_address_here
