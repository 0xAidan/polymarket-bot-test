# ============================================================
# Authentication mode
# ============================================================
# AUTH_MODE=oidc enables real SaaS login (recommended for hosted deployments).
# AUTH_MODE=legacy keeps the old shared API_SECRET flow for backwards compatibility.
AUTH_MODE=oidc

# Session cookie secret (generate 64+ random chars)
AUTH_SESSION_SECRET=replace_with_a_long_random_secret

# Auth0 OIDC settings (required when AUTH_MODE=oidc)
# AUTH0_BASE_URL must match your public app URL, e.g. https://ditto.jungle.win
AUTH0_ISSUER_BASE_URL=https://your-tenant.us.auth0.com
AUTH0_BASE_URL=https://ditto.jungle.win
AUTH0_CLIENT_ID=your_auth0_client_id
AUTH0_CLIENT_SECRET=your_auth0_client_secret

# Session timing controls (hours)
AUTH_SESSION_ABSOLUTE_DURATION_HOURS=168
AUTH_SESSION_ROLLING_DURATION_HOURS=24

# API Authentication (protects your bot's dashboard and API endpoints)
# If set, all /api/* requests require "Authorization: Bearer <API_SECRET>" header.
# The dashboard will prompt for this token on page load.
# If left empty, the API is open (a warning will be printed on startup).
# Legacy mode only. Prefer OIDC settings above for production SaaS usage.
API_SECRET=your_api_secret_here
# Production safety guard (recommended true on hosted deployments)
# When true and API_SECRET is missing, server startup fails closed.
REQUIRE_API_SECRET=true

# Your wallet private key (NEVER commit this to git!)
PRIVATE_KEY=your_private_key_here

# ============================================================
# Polymarket Builder API credentials (REQUIRED for trading!)
# ============================================================
# Without these, your orders WILL BE BLOCKED by Cloudflare.
# Get these from: https://polymarket.com/settings?tab=builder
# 
# 1. Go to polymarket.com and log in
# 2. Click your profile -> Settings -> Builder tab
# 3. Create new API keys
# 4. Copy the values below
POLYMARKET_BUILDER_API_KEY=your_builder_api_key_here
POLYMARKET_BUILDER_SECRET=your_builder_secret_here
POLYMARKET_BUILDER_PASSPHRASE=your_builder_passphrase_here

# Polymarket API endpoints (use direct URLs with Builder credentials)
POLYMARKET_CLOB_API_URL=https://clob.polymarket.com
POLYMARKET_DATA_API_URL=https://data-api.polymarket.com
POLYMARKET_GAMMA_API_URL=https://gamma-api.polymarket.com

# Polygon RPC endpoint (use a reliable provider like Alchemy, Infura, or QuickNode)
POLYGON_RPC_URL=https://polygon-rpc.com

# ============================================================
# Signature Type - IMPORTANT! Must match how you created your Polymarket account
# ============================================================
# 0 = EOA wallet (pure EOA, no proxy wallet - rare, only if you NEVER see a proxy address)
# 1 = Magic Link / Email signup (you signed up via email, Polymarket created internal key)
# 2 = Browser Wallet + Proxy (you connected MetaMask/Rabby AND Polymarket shows a "Proxy Wallet")
#
# HOW TO KNOW WHICH TO USE:
# 1. Go to polymarket.com/settings
# 2. Look for "Proxy Wallet" address
# 3. If you see a proxy address different from your MetaMask → Use type 2
# 4. If you signed up via email/Magic Link → Use type 1
# 5. If no proxy wallet exists (rare) → Use type 0
#
# If you're getting "invalid signature" errors, you likely have the wrong type!
POLYMARKET_SIGNATURE_TYPE=0

# Funder/Proxy Wallet Address - IMPORTANT FOR BALANCE DISPLAY!
# This is where Polymarket holds your USDC. To find it:
# 1. Go to polymarket.com/settings
# 2. Look for "Proxy Wallet" or "Trading Wallet" address
# 3. Copy that address here
# 
# If you see $0 balance in the dashboard but have funds on Polymarket,
# you likely need to set this to your proxy wallet address.
# 
# You can also find it by going to polygonscan.com with your EOA address
# and looking for token transfers to your Polymarket account.
POLYMARKET_FUNDER_ADDRESS=0x_your_proxy_wallet_address_here

# Server configuration (default 3001 in config)
PORT=3001

# Monitoring interval in milliseconds (how often to check for new trades)
MONITORING_INTERVAL_MS=15000

# Optional: Database for tracking (using JSON file for simplicity)
DATA_DIR=./data

# Optional: Storage backend (json | sqlite). Default: json
STORAGE_BACKEND=json

# Optional: Master password for encrypted wallet/keystore unlock (multi-wallet mode)
MASTER_PASSWORD=

DISCOVERY_V3=false
